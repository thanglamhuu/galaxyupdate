SELECT     dbo.TH_BAOGIA.A_TH_BAOGIA, 'VPAR' + CONVERT(NVARCHAR, dbo.TH_BAOGIA.A_TH_BAOGIA) AS MABAOGIA, dbo.TH_BAOGIA.SOBAOGIA, 
                      dbo.TH_BAOGIA.TENBAOGIA AS SUBJECT, dbo.TH_BAOGIA.TIEUDEBAOGIA AS GIOITHIEU, dbo.TH_BAOGIA.CAUCHAO, 
                      dbo.TH_BAOGIA.GHICHUBAOGIA, dbo.TH_BAOGIA.NGAYBAOGIA, dbo.TH_BAOGIA.NGAYHIEULUC, dbo.DM_DTTC.TENDTTC, 
                      dbo.TH_HOSOKHACHHANG.MAKH, dbo.TH_HOSOKHACHHANG.TENKHACHHANG, dbo.TH_HOSOKHACHHANG.DIACHI, dbo.TH_HOSOKHACHHANG.FAX, 
                      dbo.TH_HOSOKHACHHANG.DIENTHOAI1 AS DIENTHOAI, dbo.TH_SANPHAM.MASANPHAM, dbo.TH_SANPHAM.TENSANPHAM, 
                      dbo.TH_BAOGIA_SP.SOLUONG, dbo.TH_BAOGIA_SP.DONGIA, dbo.TH_BAOGIA_SP.THUEVAT, dbo.TH_BAOGIA_SP.CHIETKHAU, 
                      dbo.TH_BAOGIA_SP.THANHTIEN, dbo.TH_HOSOKHACHHANG.EMAIL, dbo.DM_TINHTHANHPHO.TENTINHTHANHPHO, 
                      dbo.DM_QUANHUYEN.TENQUANHUYEN, dbo.DM_LOAITIEN.TENLOAITIEN, dbo.TH_BAOGIA.TYGIAVND, dbo.TH_BAOGIA.TONGTIENCK, 
                      dbo.TH_BAOGIA.TONGTIENBAOGIA, CONVERT(NVARCHAR, 2000) AS Expr1, CONVERT(NVARCHAR, 200) AS TENCTYCHUQUAN, CONVERT(NVARCHAR, 
                      200) AS DIACHICTYCHUQUAN, CONVERT(NVARCHAR, 200) AS DIENTHOAICTYCHUQUAN, CONVERT(NVARCHAR, 200) AS FAXCTYCHUQUAN, 
                      CONVERT(NVARCHAR, 200) AS EMAILCTYCHUQUAN, CONVERT(NVARCHAR, 200) AS WEBCTYCHUQUAN, dbo.TH_BAOGIA.CHIETKHAUPT, 
                      dbo.DM_DTTC.DIDONG, dbo.DM_DTTC.EMAIL AS EMAILDTTC, dbo.DM_CHUCVU.TENCHUCVU, dbo.DM_PHONGBAN.TENPHONGBAN, 
                      CONVERT(IMAGE, NULL) AS LOGO, dbo.DM_DONVITINH.TENDONVITINH, dbo.TH_SANPHAM.DACTINHKYTHUAT AS DACTINHKT, 
                      dbo.TH_SANPHAM.IMAGE AS HINH, dbo.DM_NHOMSANPHAM.TENNHOMSANPHAM, dbo.DM_HANGSANXUAT.TENHANGSANXUAT, 
                      dbo.DM_NUOCSANXUAT.TENNUOCSANXUAT, CONVERT(NVARCHAR, N'') AS TIENTIENGVIET, CONVERT(NVARCHAR, N'') AS TIENTIENGANH, 
                      dbo.TH_SANPHAM.SOTHANGBAOHANH, CONVERT(NVARCHAR, N'') AS TENDTDTC
FROM         dbo.DM_NHOMSANPHAM RIGHT OUTER JOIN
                      dbo.TH_SANPHAM LEFT OUTER JOIN
                      dbo.DM_HANGSANXUAT ON dbo.TH_SANPHAM.ID_HANGSANXUAT = dbo.DM_HANGSANXUAT.ID_HANGSANXUAT LEFT OUTER JOIN
                      dbo.DM_NUOCSANXUAT ON dbo.TH_SANPHAM.ID_NUOCSANXUAT = dbo.DM_NUOCSANXUAT.ID_NUOCSANXUAT ON 
                      dbo.DM_NHOMSANPHAM.ID_NHOMSANPHAM = dbo.TH_SANPHAM.ID_NHOMSANPHAM LEFT OUTER JOIN
                      dbo.DM_DONVITINH ON dbo.TH_SANPHAM.ID_DONVITINH = dbo.DM_DONVITINH.ID_DONVITINH RIGHT OUTER JOIN
                      dbo.TH_BAOGIA_SP ON dbo.TH_SANPHAM.A_SANPHAM = dbo.TH_BAOGIA_SP.J_SANPHAM RIGHT OUTER JOIN
                      dbo.DM_CHUCVU RIGHT OUTER JOIN
                      dbo.DM_DTTC LEFT OUTER JOIN
                      dbo.DM_PHONGBAN ON dbo.DM_DTTC.JD_PHONGBAN = dbo.DM_PHONGBAN.ID_PHONGBAN ON 
                      dbo.DM_CHUCVU.ID_CHUCVU = dbo.DM_DTTC.JD_CHUCVU RIGHT OUTER JOIN
                      dbo.TH_BAOGIA LEFT OUTER JOIN
                      dbo.DM_LOAITIEN ON dbo.TH_BAOGIA.ID_LOAITIEN = dbo.DM_LOAITIEN.ID_LOAITIEN ON 
                      dbo.DM_DTTC.ID_DTTC = dbo.TH_BAOGIA.ID_DTTC LEFT OUTER JOIN
                      dbo.TH_BAOGIA_KH LEFT OUTER JOIN
                      dbo.DM_QUANHUYEN RIGHT OUTER JOIN
                      dbo.TH_HOSOKHACHHANG LEFT OUTER JOIN
                      dbo.DM_TINHTHANHPHO ON dbo.TH_HOSOKHACHHANG.ID_TINHTHANHPHO = dbo.DM_TINHTHANHPHO.ID_TINHTHANHPHO ON 
                      dbo.DM_QUANHUYEN.ID_QUANHUYEN = dbo.TH_HOSOKHACHHANG.ID_QUANHUYEN ON 
                      dbo.TH_BAOGIA_KH.J_HOSOKHACHHANG = dbo.TH_HOSOKHACHHANG.A_HOSOKHACHHANG ON 
                      dbo.TH_BAOGIA.A_TH_BAOGIA = dbo.TH_BAOGIA_KH.J_TH_BAOGIA ON dbo.TH_BAOGIA_SP.J_TH_BAOGIA = dbo.TH_BAOGIA.A_TH_BAOGIA
